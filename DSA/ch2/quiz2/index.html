  3. Short Answer: Provide the Big-O runtime complexity of the following loops. You can assume that any function calls within the loop have an O(1) complexity. (2 points each)

a.

int count = 0;
for (int i = 0; i < MAX; ++i) {
    count += i;
}

b.

int total = 1;
for (int i = n; i > 0; --i) {
    total *= i;
}

c. 

int x = 0;
for (int i = 0; i < n; ++i) {
    for (int j = 0; j < n: ++j) {
        x += j;
    }
    x *= i;
}

d. 

int x = 0;
while (x < (n * n)) {
    performOperation(x);
}

e. 

int x = 0;
for (int i = 1; i < n; ++i) {
    x += i;
    for (int j = n * n; j > 0; --j) {
        x /= j;
    }
    x *= i;
}
  1. Short Answer: For the following functions and classes, determine their space complexity. (2 points each)

a.

void reverseArray(int arr[], int length) {
    int *copy = new int[length];
    for (int i = 0; i < length; ++i) {
        copy[i] = arr[length - 1 - i];
    }
    for (int i = 0; i < length; ++i) {
        arr[i] = copy[i];
    }
}

b.

void reverseArray(int arr[], int length) {
    for (int i = 0; i < length / 2; ++i) {
        int temp = arr[i];
        arr[i] = arr[length - 1 - i];
        arr[length - 1 - i] = temp;
    }
}

class Matrix {
public:
    Matrix(int n, int m) {
        rows = n;
        columns = m;
        matrix = new int*[n];
        for (int i = 0; i < rows; ++i) {
            matrix[i] = new int[columns];

    }
